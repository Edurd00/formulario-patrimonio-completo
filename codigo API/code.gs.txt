function doGet(e) {
  try {
    inicializarBanco();

    const parametros = e && e.parameter ? e.parameter : {};
    const acao = normalizarTexto(parametros.acao || "").toLowerCase();

    if (acao === "gerarrelatorio" || acao === "gerar_relatorio") {
      const relatorio = gerarRelatorioPorTotvs(parametros.totvs);
      return HtmlService.createHtmlOutput(
        '<script>window.top.location.href="' + relatorio.url + '";</script>' +
        '<p>Relatorio gerado. <a href="' + relatorio.url + '" target="_top">Abrir relatorio</a></p>'
      );
    }

    if (acao === "testarroteamento" || acao === "testar_roteamento") {
      return respostaSucesso("Roteamento consultado com sucesso.", diagnosticarRoteamento(parametros.regiao, parametros.regiaoChave));
    }

    if (acao === "listar_igrejas" || acao === "listarigrejas") {
      return respostaSucesso("Lista de igrejas recuperada com sucesso.", listarTodasIgrejas());
    }

    if (acao === "validar-totvs" || acao === "validartotvs") {
      const existe = buscarLinhaBasePorTotvs(parametros.totvs) !== -1;
      return respostaSucesso("Validacao concluida.", { existe: existe });
    }

    return respostaSucesso("Backend online.", {
      app: CONFIG.APP_NAME,
      rotas: ["cadastrar", "gerarRelatorio", "exportarPdf"]
    });
  } catch (erro) {
    registrarErro("doGet", erro);
    return HtmlService.createHtmlOutput("Erro ao gerar relatorio: " + mensagemErro(erro));
  }
}

function doPost(e) {
  const lock = LockService.getScriptLock();

  try {
    lock.waitLock(30000);

    const payload = obterPayload(e);
    const acao = normalizarTexto(payload.acao || "cadastrar").toLowerCase();

    inicializarBanco();

    if (acao === "gerarrelatorio" || acao === "gerar_relatorio") {
      return respostaSucesso("Relatório gerado com sucesso.", gerarRelatorioPorTotvs(payload.totvs));
    }

    if (acao === "exportarpdf" || acao === "exportar_pdf") {
      return respostaSucesso("PDF exportado com sucesso.", exportarPdfRelatorio(payload.totvs, true));
    }

    if (acao === "testarroteamento" || acao === "testar_roteamento") {
      return respostaSucesso("Roteamento consultado com sucesso.", diagnosticarRoteamento(payload.regiao, payload.regiaoChave));
    }

    return cadastrarIgreja(payload);
  } catch (erro) {
    registrarErro("doPost", erro);
    return respostaErro(mensagemErro(erro));
  } finally {
    try {
      lock.releaseLock();
    } catch (erroLock) {
      Logger.log(erroLock);
    }
  }
}

function listarTodasIgrejas() {
  const destinos = obterDestinosConfigurados();
  let todasIgrejas = [];

  destinos.forEach(function(destino) {
    if (!destinoTemPlanilhaConfigurada(destino)) {
      return;
    }

    try {
      const planilha = abrirPlanilhaDestino(destino);
      const aba = planilha.getSheetByName(destino.base);

      if (!aba || aba.getLastRow() < 2) {
        return;
      }

      const valores = aba.getRange(2, 1, aba.getLastRow() - 1, 6).getValues();
      const igrejasDestino = valores.map(function(linha) {
        return {
          totvs: linha[0],
          regiao: linha[1],
          estadual: linha[2],
          dirigente: linha[3],
          telefone: linha[4],
          endereco: linha[5]
        };
      });

      todasIgrejas = todasIgrejas.concat(igrejasDestino);
    } catch (erro) {
      registrarErro("listarTodasIgrejas - " + destino.chave, erro);
    }
  });

  // Remover duplicatas por TOTVS (caso o mesmo TOTVS apareça em planilhas diferentes)
  const mapa = {};
  return todasIgrejas.filter(function(igreja) {
    if (mapa[igreja.totvs]) {
      return false;
    }
    mapa[igreja.totvs] = true;
    return true;
  });
}

function cadastrarIgreja(payload) {
  const dados = normalizarCadastro(payload);

  validarCadastro(dados);
  validarTotvsUnico(dados.totvs);

  salvarBaseGeral(dados);
  salvarPatrimonios(dados);
  formatarTabelasBanco();

  return respostaSucesso("Cadastro realizado com sucesso.", {
    totvs: dados.totvs
  });
}
